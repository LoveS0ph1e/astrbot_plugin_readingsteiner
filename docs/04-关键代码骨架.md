## core/everos_client.py

```python
"""EverOS REST API v1 异步客户端。
纯 HTTP 封装，输入输出皆 dict/原生类型，不依赖任何 AstrBot 类型（便于脱离 AstrBot 单测）。
契约出处：01-实证依据.md 第二部分。
"""
from __future__ import annotations
import httpx
from typing import Any


class EverOSUnavailable(Exception):
    """EverOS 不可达 / 请求失败。上层捕获后降级（跳过记忆，不阻断对话）。"""


class EverOSClient:
    def __init__(self, base_url: str, timeout: float = 30.0) -> None:
        self.base_url = base_url.rstrip("/")
        self._client = httpx.AsyncClient(timeout=timeout)

    async def close(self) -> None:
        await self._client.aclose()

    async def health(self) -> bool:
        """GET /health → data.status == 'ok'。出处：everos_client.py:30-41"""
        ...

    async def add(
        self, session_id: str, messages: list[dict[str, Any]],
        app_id: str, project_id: str,
    ) -> dict[str, Any]:
        """POST /api/v1/memory/add。
        messages 每条: {sender_id, role, timestamp(ms,int), content, [sender_name]}
        返回 data: {message_count, status: 'accumulated'|'extracted'}
        契约：01 §2.2"""
        ...

    async def flush(self, session_id: str, app_id: str, project_id: str) -> dict[str, Any]:
        """POST /api/v1/memory/flush → data.status: 'extracted'|'no_extraction'。契约：01 §2.3"""
        ...

    async def search(
        self, *, query: str,
        user_id: str | None = None, agent_id: str | None = None,
        app_id: str = "default", project_id: str = "default",
        method: str = "hybrid", top_k: int = -1,
        radius: float | None = None,
        include_profile: bool = False,
        enable_llm_rerank: bool = False,
        filters: dict | None = None,
    ) -> dict[str, Any]:
        """POST /api/v1/memory/search。
        ⚠️ user_id / agent_id 恰好设一个（否则 EverOS 返回 422）。
        返回 data: {episodes[], profiles[], agent_cases[], agent_skills[], unprocessed_messages[]}
        契约：01 §2.4"""
        ...

    async def get(
        self, *, memory_type: str,
        user_id: str | None = None, agent_id: str | None = None,
        app_id: str = "default", project_id: str = "default",
        page: int = 1, page_size: int = 20,
        sort_by: str = "timestamp", sort_order: str = "desc",
        filters: dict | None = None,
    ) -> dict[str, Any]:
        """POST /api/v1/memory/get。memory_type 仅 episode/profile/agent_case/agent_skill。
        返回 data: {<plural>[], total_count, count}。契约：01 §2.5"""
        ...

    async def _post(self, path: str, payload: dict) -> dict[str, Any]:
        """统一 POST：发请求 → raise_for_status → 取 data。
        网络/HTTP 错误统一抛 EverOSUnavailable。响应包络见 01 §2.1。"""
        try:
            resp = await self._client.post(f"{self.base_url}{path}", json=payload)
            resp.raise_for_status()
        except httpx.HTTPError as e:
            raise EverOSUnavailable(str(e)) from e
        body = resp.json()
        if "error" in body:
            raise EverOSUnavailable(body["error"].get("message", "unknown"))
        return body.get("data", {})
```

"""身份解析：AstrBot event → EverOS 身份元组。隔离正确性的命门。

# ════════════ 铁律（违反即重蹈 §15.9 跨用户串线）════════════
# 铁律 1：user_id 必须来自 event 的真实发送者 QQ 号，不接受外部/LLM 传参。
# 铁律 2：取不到有效 user_id 时返回 None，调用方跳过记忆——绝不回退 "default"。
# 铁律 3：检索只用单一 user_id，绝不轮询 [uid, "default"]。
# 反面教材：astrbot_plugin_everos_integration/tools/everos_tools.py:65,223,226-227
# ═══════════════════════════════════════════════════════════
"""
from __future__ import annotations
from dataclasses import dataclass
from astrbot.api.event import AstrMessageEvent


@dataclass
class Identity:
    user_id: str          # 发送者 QQ 号（EverOS sender_id/user_id 索引键）
    session_id: str       # = event.unified_msg_origin
    app_id: str           # 含人格隔离后缀（若在白名单）
    project_id: str
    is_group: bool
    bot_id: str           # assistant sender_id


def resolve(event: AstrMessageEvent, config) -> Identity | None:
    """从 event 确定性解析身份。取不到 user_id 返回 None。

"""注入：检索结果 → 记忆文本 → 改写 ProviderRequest。范式出处：01 §3.3。"""
from __future__ import annotations
from astrbot.api.provider import ProviderRequest


def build_text(profiles: list[dict], episodes: list[dict], config) -> str:
    """拼装注入文本（画像段 + 情景段）。结构见 02-架构设计.md §3.1。
    - profiles[].profile_data 渲染为「用户长期印象」（恒注入）
    - episodes[] 按 score 取 top-n，渲染为「相关情景记忆」
    - 都为空返回 ''（调用方据此跳过注入）"""
    prefix = config.get("memory_prefix", "<EverOS Memory>")
    suffix = config.get("memory_suffix", "</EverOS Memory>")
    parts = []
    if profiles:
        parts.append("【用户长期印象】\n" + _render_profile(profiles[0]))
    if episodes:
        parts.append("【相关情景记忆】\n" + _render_episodes(episodes))
    if not parts:
        return ""
    return f"{prefix}\n" + "\n\n".join(parts) + f"\n{suffix}"


def inject(req: ProviderRequest, text: str, target: str, position: str) -> None:
    """改写 req。target: system_prompt|user_prompt; position: prepend|append。
    出处：Mnemosyne memory_operations.py:1029-1046。"""
    if not text:
        return
    if target == "system_prompt":
        cur = req.system_prompt if isinstance(req.system_prompt, str) else ""
        req.system_prompt = (text + cur) if position == "prepend" else (cur + text)
    else:  # user_prompt
        cur = req.prompt if isinstance(req.prompt, str) else ""
        req.prompt = (text + "\n" + cur) if position == "prepend" else (cur + "\n" + text)

## main.py（接线骨架）

```python
from astrbot.api import AstrBotConfig, logger
from astrbot.api.event import AstrMessageEvent, filter
from astrbot.api.provider import LLMResponse, ProviderRequest
from astrbot.api.star import Context, Star, register

from .core import identity as identity_mod
from .core import injection, archiving, visibility
from .core.everos_client import EverOSClient, EverOSUnavailable
from .core.constants import DEFAULT_BASE_URL


@register("astrbot_plugin_everos", "<作者>",
          "基于 EverOS 自进化记忆引擎的长期记忆插件", "v0.1.0", "<仓库URL>")
class EverOSMemoryPlugin(Star):
    def __init__(self, context: Context, config: AstrBotConfig):
        super().__init__(context)
        self.config = config
        self.context = context
        self.client: EverOSClient | None = None
        self._flush_policy: archiving.FlushPolicy | None = None
        self._bg_task = None

    @filter.on_astrbot_loaded()
    async def on_loaded(self):
        try:
            self.client = EverOSClient(
                self.config.get("everos_base_url", DEFAULT_BASE_URL),
                self.config.get("request_timeout", 30.0))
            self._flush_policy = archiving.FlushPolicy(
                self.config.get("archive_strategy", "auto"),
                self.config.get("flush_idle_seconds", 300))
            # 起 auto-flush 后台任务（asyncio）
            ...
        except Exception as e:
            logger.error(f"[everos] 初始化失败: {e}", exc_info=True)

    @filter.on_llm_request()
    async def on_llm_request(self, event: AstrMessageEvent, req: ProviderRequest):
        try:
            if not self.config.get("enable_injection", True):
                return
            ident = identity_mod.resolve(event, self.config)
            if ident is None:                       # 铁律 2
                return
            if not await self._healthy():
                return
            data = await self.client.search(
                query=_extract_query(req), user_id=ident.user_id,
                app_id=ident.app_id, project_id=ident.project_id,
                method=self.config.get("search_method", "hybrid"),
                top_k=self.config.get("search_top_k", 5),
                include_profile=self.config.get("include_profile", True))
            profiles, episodes = data.get("profiles", []), data.get("episodes", [])
            if self.config.get("group_public_only", True) and ident.is_group:
                profiles, episodes = visibility.filter_public(profiles, episodes)
            text = injection.build_text(profiles, episodes, self.config)
            injection.inject(req, text,
                self.config.get("injection_target", "system_prompt"),
                self.config.get("injection_position", "prepend"))
        except Exception as e:
            logger.error(f"[everos] on_llm_request 失败: {e}", exc_info=True)

    @filter.on_llm_response()
    async def on_llm_response(self, event: AstrMessageEvent, resp: LLMResponse):
        try:
            if not self.config.get("enable_archiving", True):
                return
            ident = identity_mod.resolve(event, self.config)
            if ident is None:
                return
            msgs = archiving.build_messages(
                _extract_user_text(event), _extract_resp_text(resp), ident)
            await self.client.add(ident.session_id, msgs, ident.app_id, ident.project_id)
            self._flush_policy.mark_active(ident.session_id)
            if self._flush_policy.should_flush_now(ident.session_id):
                await self.client.flush(ident.session_id, ident.app_id, ident.project_id)
        except Exception as e:
            logger.error(f"[everos] on_llm_response 失败: {e}", exc_info=True)

    async def terminate(self):
        if self._bg_task:
            self._bg_task.cancel()
        if self.client:
            await self.client.close()

    async def _healthy(self) -> bool:
        try:
            return await self.client.health()
        except EverOSUnavailable:
            return False
```
